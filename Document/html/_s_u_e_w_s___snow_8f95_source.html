22 SUBROUTINE meltheat
23 
24  USE allocatearray
25  USE snowmod
26  USE data_in
27  USE snowmod
28  USE time
29  USE moist
30  USE sues_data

35  REAL(KIND(1d0)):: Watfreeze,& !State of snow on ith surface, total energy exchange (not exported!)
36  cw=4190!,ci=2090 !Specific heat capacity of water
37 
38  !Initialize snow variables
39  snowcalcswitch=0
40  statefraction=0
41  snowcoverforms=0
42  qm_melt=0
43  qm_freezstate=0
44  qm_rain=0
45  freezmelt=0
46  freezstate=0
47  freezstatevol=0
48  rainonsnow = 0
49 
50  !Initialize mechanical snow removal
51  snowremoval(pavsurf)=0
52  snowremoval(bldgsurf)=0
53 
54  !=========================================================================================
55  DO is=1,nsurf !Go each surface type through
56  IF (sfr(is)/=0) THEN !If surface type existing,
57 
58  IF (snowpack(is)>0) THEN !If snowpack existing, calculate meltwater related water flows
59 
60  snowdepth(is) = (snowpack(is)/1000)*waterdens/snowdens(is) !Snow depth in m
61 
62  !Calculate meltwater related water flows with hourly degree-day method.
63 
64  !These are for snow melting
65  IF (temp_c>=0) THEN
66  IF (qn1_ind_snow(is)<0) THEN
67  mw_ind(is) = tempmeltfact*temp_c !(mm Câˆ’1 hâˆ’1)*(C) = in mm h-1
68  ELSE
69  mw_ind(is) = radmeltfact*(qn1_ind_snow(is)) !(mm m2 Wâˆ’1 hâˆ’1)*(W m-2)= mm h-1 ??
70  ENDIF
71 
72  ELSE !Freezing equations
73  adjmeltfact=1 !Relationship between the temperature melt and freezing factors
74  mw_ind(is) = tempmeltfact*temp_c*adjmeltfact ! in mm h-1
75  ENDIF
76 
77  !Previous equation give the hourly values, divide these with the timestep number
78  mw_ind(is) = mw_ind(is)/nsh_real
79 
80  IF (mw_ind(is)>snowpack(is)) mw_ind(is) = snowpack(is)!Limited by the previous timestep snowpack
81 
82  !-----------------------------------------------------
83  ! Heat consumed to snowmelt/refreezing within Tstep.
84  ! Converted from mm nsh-1 to mm nsh-1 and to m s-1
85  qm_melt(is) = waterdens*((mw_ind(is)/tstep_real)/1000)*(lvs_j_kg-lv_j_kg)
86 
87  !If melt is negative this means freezing water in the snowpack
88  IF (mw_ind(is)<0) THEN
89 
90  freezmelt(is) = -mw_ind(is) !Save this to variable FreezMelt
91  mw_ind(is) = 0
92 
93  !Freezing water cannot exceed meltwater store
94  IF (freezmelt(is)>meltwaterstore(is)) freezmelt(is) = meltwaterstore(is)
95 
96  !Recalculate melt related energy
97  qm_melt(is) = waterdens*((-freezmelt(is)/tstep_real)/1000)*(lvs_j_kg-lv_j_kg)
98  ENDIF
99 
100  !-----------------------------------------------------
101  ! If air temperature is above zero, precipitation causes advective heat to the
102  ! snowpack. Eq (23) in Sun et al., 1999
103  ! Calculation done at resolution of the model timestep
104  IF (temp_c>=preciplimit.AND.precip>0) THEN
105  qm_rain(is) = waterdens*cw*(temp_c-preciplimit)*(precip*0.001/tstep_real) !in W m-2
106  IF (qm_rain(is)<0) THEN !Can only be positive
107  qm_rain(is) = 0
108  ELSE
109  rainonsnow(is) = precip !Save information on the rain on snow event
110  ENDIF
111  ENDIF
112 
113  ENDIF !End if snowpack
114 
115  !=================================================================
116 
117  !Freeze surface water state if cold enough.
118  IF (tsurf_ind(is)<0.AND.state(is)>0) THEN
119 
120  snowcalcswitch(is)=1 !If water on ground this forms ice and snow calculations are made
121 
122  !Other surfaces than water treated first
123  IF (is/=watersurf) THEN
124 
125  !FreezState(is) = state(is)
126  !Previously all state could freeze in 5-min timestep. Now we calculate how much water
127  !can freeze in a timestep based on the same temperature freezing fraction.
128  freezstate(is) = -tempmeltfact*tsurf_ind(is)/nsh_real
129 
130  !The amount of freezing water cannot be greater than the surface state
131  IF (freezstate(is)>state(is)) freezstate(is) = state(is)
132 
133  IF (snowpack(is)==0.OR.snowfrac(is)==0) THEN !Snowpack forms
134  freezstatevol(is) = freezstate(is)
135  ELSE !There is snow already on ground
136  freezstatevol(is) = freezstate(is)*(1-snowfrac(is))/snowfrac(is)
137  ENDIF
138 
139  ! If the amount of freezing water is very small and there is state left to the ground
140  ! no freezing of water will take place
141  IF (freezstatevol(is)<0.00000000001.AND.freezstate(is)<state(is)) THEN
142  freezstate(is) = 0
143  freezstatevol(is) = 0
144  ENDIF
145 
146  !Calculate the heat exchange in W m-2
147  qm_freezstate(is) = -waterdens*(freezstate(is)/tstep_real/1000)*(lvs_j_kg-lv_j_kg)
148 
149  !Water surface separately
150  ELSE
151  !Calculate average value how much water can freeze above the water areas
152  !Equation is -hA(T-T0) = rhoV(Cp+dT +Lf) in 5-min timestep
153  !h=convective heat trasnfer,A, area of water,rwo water density,V volume, dT temperature difference
154  !before and end of the 5-min period. dT equals zero, h=100 and when multiplied with Area, the equation
155  !simplyfies to the this. LJ 14 July 2015
156  watfreeze = 100*(0-temp_c)/(waterdens*(lvs_j_kg-lv_j_kg))
157  freezstate(is) = watfreeze
158  qm_freezstate(is) = -waterdens*(watfreeze/tstep_real/1000)*(lvs_j_kg-lv_j_kg)
159  ENDIF
160 
161  ENDIF
162 
163  !======================================================================
164  ! Define if any snowmelt calculations are made: snowpack existing,
165  ! freezing occuring on ground or from precip
166  IF (is/=watersurf) THEN
167  IF (snowpack(is)>0.OR.(precip>0.AND.tsurf_ind(is)<0)) THEN
168  snowcalcswitch(is)=1
169  ENDIF
170  ELSE !Water surface separately
171  IF (snowpack(watersurf)>0.OR.freezstate(watersurf)>0) THEN
172  snowcalcswitch(watersurf)=1
173  ENDIF
174  ENDIF
175 
176  !Update snow density of each surface
177  IF (precip>0.AND.tsurf_ind(is)<0.AND.snowpack(is)>0) THEN
178  snowdens(is) = snowdens(is)*snowpack(is)/(snowpack(is)+precip)+snowdensmin*precip/(snowpack(is)+precip)
179  ENDIF
180 
181  !Weighted variables for the whole area
182  mwh = mwh + mw_ind(is)*sfr(is)*snowfrac(is) !Snowmelt
183  fwh = fwh + freezmelt(is)*sfr(is)*snowfrac(is) !Freezing water
184  qm = qm + qm_melt(is)*sfr(is)*snowfrac(is) !Energy consumed to the melt/freezing.
185  qmrain = qmrain + qm_rain(is)*sfr(is)*snowfrac(is) !Rain on snow
186  qmfreez=qmfreez+deltaqi(is)*sfr(is)*snowfrac(is)+qm_freezstate(is)*sfr(is)*(1-snowfrac(is)) !Freezing water
187  ENDIF
188 
189  ENDDO !End surface type
190 
191  !Update snow albedo to its maximum value if precipitation exists
192  IF (precip>0.AND.sum(snowpack)>0.AND.temp_c<0) THEN
193 
194  cumsnowfall=cumsnowfall + precip
195 
196  IF (cumsnowfall>preciplimitalb) THEN
197  snowalb=snowalbmax
198  cumsnowfall=0
199  ENDIF
200  ELSE
201  cumsnowfall=0
202  ENDIF
203 
204 
205 END SUBROUTINE meltheat

210 SUBROUTINE snowcalc
211  !Calculation of snow and water balance on 5 min timestep. Treats snowfree and snow covered
212  !areas separately. Weighting is taken into account in the overall values.
213  !Last modified:
214  ! LJ in 6 May 2015 - Modified to run with timestep
215  ! HCW 06 Mar 2015 - Unused variable 'i' removed.
216  ! HCW 26 Jan 2015 - Added weekday/weekend option for snow clearing profiles
217  ! LJ in 24 May 2013
218  !========================================================================
219 
220  USE allocatearray
221  USE defaultnotused
222  USE data_in
223  USE gis_data
224  USE mod_k
225  USE mod_z
226  USE moist
227  USE snowmod
228  USE sues_data
229  USE thresh
230  USE time
231 
232  IMPLICIT NONE
233 
234  REAL(KIND(1d0)):: Evap_SUEWS_Snow,&
235  MeltExcess,& !Excess melt water that needs to leave snowpack
236  snowTotInit,&
237  EvPart,&
238  runoffTest,&
239  snowFracFresh1,& !Snow fraction for newly formed snowpack
240  snowFracFresh2,& !Snow fraction for newly formed snowpack from state only
241  snowFracOld,&
242  FWC !Water holding capacity of snow in mm
243  REAL(KIND(1d0)):: SnowDepletionCurve
244 
245  INTEGER:: iu !1=weekday OR 2=weekend
246  !========================================================================
247  !Initialize variables for the calculation of water storages and evaporation
248 
249  ! Use weekday or weekend snow clearing profile
250  iu=1 !Set to 1=weekday
251  IF(dayofweek(id,1)==1.OR.dayofweek(id,1)==7) iu=2 !Set to 2=weekend
252 
253  runoffsnow(is)=0 !Initialize for runoff caused by snowmelting
254  runoff(is)=0
255  runoffsoil(is)=0
256  chang(is)=0
257  changsnow(is)=0
258  runofftest=0
259  snowtosurf(is)=0
260  evpart=0
261  ev=0
262  snowfracfresh1=0
263  snowfracfresh2=0
264  snowfracold=0
265 
266  !Initial snowpack + meltwater in it
267  snowtotinit=snowpack(is)+meltwaterstore(is)
268 
269  !Calculate water holding capacity (Jin et al. 1999)
270  IF (snowdens(is)>=200) THEN
271  waterholdcapfrac=crwmin
272  ELSE
273  waterholdcapfrac=crwmin+(crwmax-crwmin)*(200-snowdens(is))/200
274  ENDIF
275 
276  !======================================================================
277  ! Calculate evaporation from snowpack and snow free surfaces (in mm)
278  IF (snowfrac(is)<1) CALL evap_suews !ev and qe for snow free surface out
279 
280  IF (snowfrac(is)>0) THEN
281  ev_snow(is) = evap_suews_snow(qm_melt(is),qm_rain(is),lvs_j_kg,avdens,avrh,press_hpa,temp_c,rasnow,&
282  psyc_hpa,tstep,avcp,sice_hpa)
283  ENDIF
284 
285  !If not enough water for evaporation in impervious surfaces,
286  !evaporation is taken from pervious surfaces
287  IF (is>2) THEN
288  IF (pervfraction/=0) THEN
289  evpart=(surplusevap(pavsurf)*sfr(pavsurf)+surplusevap(bldgsurf)*sfr(bldgsurf))/pervfraction
290  ENDIF
291  ENDIF
292 
293 
294  !============================================================================
295  !Water surface is treated separately
296  IF (is==watersurf.AND.sfr(watersurf)>0) GO TO 606
297 
298  !The calculations are divided into 2 main parts
299  ! 1) Surface is fully covered with snow at the beginning of the time step
300  ! 2) Surface is not fully covered with snow but rather part is snow free OR
301  ! surface not orginally covered with snow, but the snow forms at the current timestep
302 
303  !1)------------------------------------------------------------------
304  ! ------------------------------------------------------------------
305  IF (snowpack(is)>0.AND.snowfrac(is)==1) THEN
306 
307  ev_snow(is)=ev_snow(is)+evpart !Evaporation surplus
308 
309  !(Snowfall per interval+freezing of melt water and surface state) - (meltwater+evaporation from snowpack)
310  changsnow(is)=(precip+freezmelt(is))-(mw_ind(is)+ev_snow(is)) !Calculate change in snowpack (in mm)
311 
312  !If rain on snow event, add this water to meltwaterstore
313  IF (rainonsnow(is)>0) THEN
314  changsnow(is)=changsnow(is)-precip
315  meltwaterstore(is) = meltwaterstore(is)+rainonsnow(is)
316  ENDIF
317 
318  snowpack(is)=snowpack(is)+changsnow(is) !Update snowpack
319 
320  !---------If snowpack exists after the state calculations
321  IF (snowpack(is)>0) THEN
322 
323  !Add melted water to meltstore and freeze water according to freezMelt(is)
324  meltwaterstore(is) = meltwaterstore(is) + mw_ind(is) - freezmelt(is)
325 
326  !Calculate water holding capacity (FWC: Valeo and Ho, 2004) of the snowpack
327  fwc = waterholdcapfrac*snowpack(is)
328 
329  !If FWC is exceeded, excess meltwater (MeltExcess) will leave from the snowpack
330  IF (meltwaterstore(is)>=fwc) THEN
331  meltexcess = 0 !Initialize the excess meltwater
332  meltexcess = meltwaterstore(is)-fwc !Calculate the exceess water
333  meltwaterstore(is) = fwc !Update the meltwaterstore to the maximum it can hold
334  runoffsnow(is) = runoffsnow(is) + meltexcess
335  ENDIF
336 
337  !At the end of the hour calculate possible snow removal
338  IF (snowprof(it,iu)==1.AND.is<3.AND.(imin==(nsh_real-1)/nsh_real*60)) CALL snowrem
339 
340  !----------If snowPack is negative, it melts at this timestep
341  ELSEIF (snowpack(is)<0) THEN
342 
343  !If freezing meltwater inside this timestep, remove it from the MeltWaterStore
344  meltwaterstore(is)=meltwaterstore(is)-freezmelt(is)+mw_ind(is)+snowpack(is)
345  snowpack(is)=0.0 !Set the snow pack and snow
346  snowfracold=1
347  snowfrac(is)=0
348  snowdens(is)=0
349 
350  IF (meltwaterstore(is)<0) THEN !Not enough water in the meltwater store,
351  ev_snow(is)=ev_snow(is)+meltwaterstore(is) !evaporation from snow is decreased.??
352  IF (ev_snow(is)<0) ev_snow(is)=0
353  changsnow(is)=changsnow(is)+meltwaterstore(is)
354  meltwaterstore(is)=0
355  ELSE
356  chang(is)=meltwaterstore(is) !Meltwater goes to surface state as no snow exists anymore
357  state(is)=state(is)+chang(is)
358  meltwaterstore(is)=0
359  ENDIF
360  ENDIF !snowpack negative or positive
361 
362 
363  !2)------Surface not fully covered with snow-------------------------------------------
364  ! ------------------------------------------------------------------------------------
365  ELSEIF (snowfrac(is)<1) THEN
366 
367  !Snow calculations: snowpack can either exist or form at the current timestep
368  IF (snowpack(is)>0) THEN
369  ev_snow(is)=ev_snow(is)+evpart !Evaporation surplus
370 
371 
372  !----Snowpack water balance for the whole surface area. In reality snow depth = snowPack/snowFrac(is)
373  !(Snowfall per interval+freezing of melt water and surface state) - (meltwater+evaporation from snowpack)
374  changsnow(is)=(precip+freezmelt(is)+freezstatevol(is))-(mw_ind(is)+ev_snow(is)) !Calculate change in snowpack (in mm)
375 
376  !If rain on snow event, add this water to meltwaterstore
377  IF (rainonsnow(is)>0) THEN
378  changsnow(is)=changsnow(is)-precip
379  meltwaterstore(is) = meltwaterstore(is)+rainonsnow(is)
380  ENDIF
381  snowpack(is)=snowpack(is)+changsnow(is)
382 
383 
384  !The fraction of snow will update when:
385  !a) Surface state is dry but precipitation occurs =1
386  !b) There is both precipitation and all surface state freezes =1
387  !c) No precipitation but all state freezes at a single timestep =2
388  !d) Part of the surface freezes
389  IF (precip>0.AND.freezstate(is)==state(is)) THEN !both a) and b)
390  snowfracfresh1=1
391  ELSEIF (precip==0.AND.freezstate(is)>0.AND.freezstate(is)==state(is)) THEN
392  snowfracfresh1=1
393 
394  !snowFracFresh1=SnowDepletionCurve(is,snowPack(is),snowD(is))
395  !if (snowFracFresh1<0.001) snowFracFresh1=0.001
396  ELSEIF (freezstate(is)>0.AND.freezstate(is)<state(is)) THEN !This if not all water freezes
397  snowfracfresh1=0.95 !Now this fraction set to something close to one. Should be improved in the future at some point
398  !if (is==1)then
399  ! write(*,*) id,it,imin,snowfrac(is),FreezState(is),state(is)
400  ! pause
401  !endif
402  ENDIF
403 
404  !Snowpack can also form at the current timestep (2). If this forms purely from snowfall or/and all water at surface freezes,
405  !the whole surface will be covered with snow. If there is water on ground this snowfall can immediately melt
406  !and in this case the snow fraction is not necessarily 1 but its information is saved to snowFracFresh that
407  !is taken into account in snow fraction after calculation of state.
408  ELSEIF (snowpack(is)==0.AND.tsurf_ind(is)<0) THEN
409 
410  !The fraction of snow will get a value of 1 (ie full snow cover):
411  !Surface state is dry but precipitation occurs, no precipitation but all state freezes at a single timestep,
412  !There is both precipitation and all surface state freezes
413  IF ((precip>0.AND.state(is)==0).OR.(precip==0.AND.freezstate(is)==state(is)).OR.&
414  (precip>0.AND.freezstate(is)==state(is))) THEN
415 
416  !ev=ev+EvPart
417  changsnow(is)=precip+freezstatevol(is)
418  snowpack(is)=snowpack(is)+changsnow(is) !Update snowpack
419 
420  snowfracfresh1=1
421  icefrac(is)=freezstate(is)/(freezstate(is)+precip)
422  snowdens(is)=snowdensmin
423  ENDIF
424 
425  IF (freezstate(is)>0.AND.freezstate(is)<state(is)) THEN
426 
427  changsnow(is)=precip+freezstatevol(is)
428  snowpack(is)=snowpack(is)+changsnow(is) !Update snowpack
429  snowfracfresh2=0.95 !Now this fraction set to something close to one. Should be improved in the future at some point
430 
431  !snowFracFresh2=SnowDepletionCurve(is,snowPack(is),snowD(is))
432  !if (snowFracFresh2<0.001) snowFracFresh2=0.001
433  icefrac(is)=1
434  snowdens(is)=snowdensmin
435  !write(*,*) 2,is,id,it,imin,snowfrac(is),FreezState(is),state(is),state(is)+Precip
436  !pause
437 
438  ENDIF
439  ENDIF
440 
441  !---------If snowpack exists after the state calculations
442  IF (snowpack(is)>0) THEN
443 
444  !Add melted water to meltstore and freeze water according to freezMelt(is)
445  meltwaterstore(is) = meltwaterstore(is) + mw_ind(is) - freezmelt(is)
446 
447  !Calculate water holding capacity (FWC: Valeo and Ho, 2004) of the snowpack
448  fwc = waterholdcapfrac*snowpack(is)
449 
450  !If FWC is exceeded, excess meltwater (MeltExcess) will leave from the snowpack
451  IF (meltwaterstore(is)>=fwc) THEN
452  meltexcess = 0 !Initialize the excess meltwater
453  meltexcess = meltwaterstore(is)-fwc !Calculate the exceess water
454  meltwaterstore(is) = fwc !Update the meltwaterstore to the maximum it can hold
455 
456  !If the fraction of snow is greater than 0.8 or if the surface is is buildings,
457  !the excess water will directly go to runoff. Otherwise it will flow to the
458  !snow free area via SnowToSurf(is)
459  IF ((snowfrac(is)>0.9.AND.is/=bldgsurf).OR.(is==bldgsurf)) THEN
460  runoffsnow(is) = runoffsnow(is) + meltexcess
461  ELSE
462  snowtosurf(is) = snowtosurf(is) + meltexcess*snowfrac(is)/(1-snowfrac(is))
463  ENDIF
464  ENDIF
465 
466  !At the end of the hour calculate possible snow removal
467  IF (snowprof(it,iu)==1.AND.is<3.AND.(imin==(nsh_real-1)/nsh_real*60)) CALL snowrem
468 
469  !----------If snowPack is negative, it melts at this timestep
470  ELSEIF (snowpack(is)<0) THEN
471 
472  !If freezing meltwater inside this timestep, remove it from the MeltWaterStore
473  meltwaterstore(is)=meltwaterstore(is)-freezmelt(is)+mw_ind(is)+snowpack(is)
474 
475  snowpack(is)=0.0 !Set the snow pack and snow
476  snowfracfresh1=0
477  snowfracfresh2=0
478  snowdens(is)=0
479 
480  IF (meltwaterstore(is)<0) THEN !Not enough water in the meltwater store,
481  ev_snow(is)=ev_snow(is)+meltwaterstore(is) !evaporation from snow is decreased.??
482  IF (ev_snow(is)<0) ev_snow(is)=0
483  changsnow(is)=changsnow(is)+meltwaterstore(is)
484  meltwaterstore(is)=0
485  ELSE
486  snowtosurf(is)=snowtosurf(is)+meltwaterstore(is)*snowfrac(is)/(1-snowfrac(is))
487  meltwaterstore(is)=0
488  ENDIF
489  ENDIF !snowpack negative or positive
490 
491 
492  !--------
493  !Next the snow free surface (3). Calculations only done if snowfraction is smaller than 1
494  IF ((is==pavsurf.OR.is==bldgsurf).AND.snowfrac(is)<1) THEN !Impervious surfaces (paved, buildings)
495 
496  !Surface store update. If precipitation is greater than the threshold, the exceeding water
497  !goes directly to runoff
498  IF (precip>ipthreshold_mmhr/nsh_real) THEN
499  !runoff = runoff + (precipitation+water from the snow surface+water from other surfaces-the thereshold limit)
500  runoff(is)=runoff(is)+(precip+snowtosurf(is)+addwater(is)-ipthreshold_mmhr/nsh_real)
501  chang(is)=ipthreshold_mmhr/nsh_real-(drain(is)+ev+freezstate(is))
502  ELSE
503  !Add precip and water from other surfaces and remove drainage, evap and freezing of state
504  chang(is)=precip+snowtosurf(is)+addwater(is)-(drain(is)+ev+freezstate(is))
505  ENDIF
506 
507  state(is)=state(is)+chang(is) !Change in state (for whole surface area areasfr(is))
508 
509  !Add water from impervious grids
510  ! Check sfr/=0 added HCW 08 Dec 2015
511  IF (is==pavsurf.AND.sfr(pavsurf)/=0) state(is)=state(is)+(addimpervious)/sfr(pavsurf)
512 
513  runoff(is)=runoff(is)+drain(is)*addwaterrunoff(is) !Drainage (not flowing to other surfaces) goes to runoff
514 
515  IF(state(is)<0.0) THEN !Surface state cannot be negative
516  surplusevap(is)=abs(state(is)) !take evaporation from other surfaces in mm
517  ev = ev-surplusevap(is)
518  state(is)=0.0
519  ENDIF
520 
521  ELSEIF(is>=3.AND.snowfrac(is)<1) THEN ! Pervious surfaces (conif, decid, grass unirr, grass irr)
522 
523  ev=ev+evpart
524 
525  !Change in water stores
526  IF (precip+addveg*(sfr(is)/vegfraction)>(ipthreshold_mmhr/nsh_real)) THEN !if 5min precipitation is larger than 10 mm
527  runoff(is)=runoff(is)+(precip+addveg*(sfr(is)/vegfraction)+snowtosurf(is)+addwater(is)-(ipthreshold_mmhr/nsh_real))
